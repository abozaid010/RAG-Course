# =============================================================================
# WEB FRAMEWORK & API SERVER
# =============================================================================

# FastAPI - Modern, fast web framework for building APIs
# WHY NEEDED: Provides the REST API interface for our RAG system
# FEATURES: Automatic API documentation, validation, serialization, async support
# ALTERNATIVES: Flask, Django (but FastAPI is best for modern async APIs)
fastapi==0.110.0

# Uvicorn - ASGI server for running FastAPI applications
# WHY NEEDED: Serves our FastAPI application in production
# FEATURES: High performance, async support, WebSocket support
# [standard] includes extra dependencies for better performance and monitoring
uvicorn[standard]==0.29.0

# Pydantic - Data validation and settings management
# WHY NEEDED: Validates incoming API requests and serializes responses
# FEATURES: Type hints, automatic validation, serialization, documentation generation
# INTEGRATION: Deeply integrated with FastAPI for request/response handling
pydantic==2.6.4

# =============================================================================
# HTTP CLIENT & API COMMUNICATION
# =============================================================================

# HTTPX - Async HTTP client for making API calls
# WHY NEEDED: Communicates with DeepSeek's LLM API asynchronously
# FEATURES: Async/await support, HTTP/2, connection pooling, timeouts
# COMPARISON: Replaces requests library with async capabilities for better performance
httpx==0.27.0

# Tenacity - Retry library for handling transient failures
# WHY NEEDED: Makes our LLM API calls more reliable by automatically retrying failed requests
# FEATURES: Exponential backoff, retry conditions, jitter, stop conditions
# USE CASE: Handles network issues, rate limits, temporary API failures
tenacity==8.2.3

# =============================================================================
# VECTOR DATABASE & EMBEDDINGS
# =============================================================================

# ChromaDB - Open-source vector database for AI applications
# WHY NEEDED: Stores document embeddings and enables fast similarity search
# FEATURES: Vector storage, similarity search, metadata filtering, easy integration
# ROLE IN RAG: Provides the "Retrieval" component - finds relevant documents
chromadb==0.4.24

# Sentence Transformers - Pre-trained models for text embeddings
# WHY NEEDED: Converts text documents to numerical vectors for similarity search
# FEATURES: Pre-trained models, multiple languages, GPU support, easy integration
# MODEL USED: all-MiniLM-L6-v2 (lightweight, fast, good quality)
sentence-transformers==2.7.0

# =============================================================================
# ENVIRONMENT & CONFIGURATION
# =============================================================================

# python-dotenv - Environment variable management
# WHY NEEDED: Loads environment variables from .env files for configuration
# FEATURES: .env file support, environment variable loading, configuration management
# SECURITY: Keeps API keys and sensitive data out of source code
python-dotenv==1.0.1

# orjson - Fast JSON parsing and serialization
# WHY NEEDED: Improves performance of JSON operations in our API
# FEATURES: Faster than standard json library, better memory efficiency
# USE CASE: Parsing API requests, serializing responses, handling large JSON data
orjson==3.10.1

# =============================================================================
# OPTIONAL ENHANCEMENTS

# Redis - In-memory data structure store (optional)

# USE CASES: Cache LLM responses, rate limiting, session storage
